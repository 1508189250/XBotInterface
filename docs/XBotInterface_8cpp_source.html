194 bool XBot::XBotInterface::init(const std::string &path_to_cfg, AnyMapConstPtr any_map)
195 {
196  // store path to config
197  _path_to_cfg = path_to_cfg;
198  // parse the YAML file to initialize internal variables
199  parseYAML(path_to_cfg);
200  // initialize the model
201  if (!_XBotModel.init(_urdf_path, _srdf_path, _joint_map_config)) {
202  printf("ERROR: model initialization failed, please check the urdf_path and srdf_path in your YAML config file.\n");
203  return false;
204  }
205  // generate the robot
206  _XBotModel.generate_robot();
207 
208  // construct class
209  _joint_num = _XBotModel.get_joint_num();
210  _urdf_string = _XBotModel.get_urdf_string();
211  _srdf_string = _XBotModel.get_srdf_string();
212  _XBotModel.get_enabled_joint_ids(_ordered_joint_id);
213  _XBotModel.get_enabled_joint_names(_ordered_joint_name);
214  _ordered_chain_names = _XBotModel.get_ordered_chain_names();
215 
216  // create dynamically the Kinematic Chains, FT and IMU
217  for (const std::string & chain_name : _XBotModel.get_chain_names()) {
218  XBot::KinematicChain::Ptr actual_chain = std::make_shared<KinematicChain>(chain_name,
219  _XBotModel);
220  _chain_map[chain_name] = actual_chain;
221 
222  const std::map< std::string, ForceTorqueSensor::Ptr >& ft_map = _chain_map.at(chain_name)->getForceTorqueInternal();
223  _ft_map.insert(ft_map.begin(),
224  ft_map.end());
225 
226  const std::map< std::string, ImuSensor::Ptr >& imu_map = _chain_map.at(chain_name)->getImuInternal();
227  _imu_map.insert(imu_map.begin(),
228  imu_map.end());
229 
230  for( int i = 0; i < actual_chain->getJointNum(); i++){
231  _joint_vector.push_back(actual_chain->getJointInternal(i));
232  }
233 
234  }
235 
236  // NOTE if you have disabled joint, the URDF should be updated in order to have compatibility btw robot and model
237  if (_XBotModel.getDisabledJoints().size() > 0) {
238  std::cerr << "WARNING in " << __func__ << " : disabled joint detected in the specified SRDF. URDF must be updated accordingly (we hope you did it)." << std::endl;
239  }
240 
241  // call virtual init_internal
242  bool success = init_internal(path_to_cfg, any_map);
243 
244  // after subclasses have done their work inside init_internal, compute joint number
245  _joint_num = 0;
246  for( const auto& c : _chain_map ){
247  _joint_num += c.second->getJointNum();
248  }
249 
250  // fill the joint id to eigen id
251  _ordered_joint_vector.clear();
252  _joint_name_to_eigen_id.clear();
253  _ordered_joint_id.clear();
254  _ordered_joint_name.clear();
255  int eigen_id = 0;
256 
257  for( const std::string& chain_name : getModelOrderedChainName() ) {
258  for( int i = 0; i < _chain_map.at(chain_name)->getJointNum(); i++) {
259  _joint_id_to_eigen_id[_chain_map.at(chain_name)->getJointId(i)] = eigen_id;
260  _joint_name_to_eigen_id[_chain_map.at(chain_name)->getJointName(i)] = eigen_id;
261  _ordered_joint_vector.push_back(_chain_map.at(chain_name)->getJointInternal(i));
262  _ordered_joint_id.push_back(_chain_map.at(chain_name)->getJointId(i));
263  _ordered_joint_name.push_back(_chain_map.at(chain_name)->getJointName(i));
264  eigen_id++;
265  }
266  }
267 
268  _joint_vector = _ordered_joint_vector;
269 
270  // fill FT & IMU id maps
271  for( const auto& pair : getForceTorqueInternal() ){
272  _ft_id_map[pair.second->getSensorId()] = pair.second;
273  }
274 
275  for( const auto pair : getImuInternal() ){
276  _imu_id_map[pair.second->getSensorId()] = pair.second;
277  }
278 
279  return success;
280
