34 KinematicChain::KinematicChain(const std::string &chain_name,
35  const XBot::XBotCoreModel &XBotModel) :
36  _chain_name(chain_name),
37  _XBotModel(XBotModel),
38  _is_virtual(false),
39  _joint_num(0)
40 {
41  _joint_num = _XBotModel.get_joint_num(chain_name);
42  _XBotModel.get_enabled_joints_in_chain(chain_name, _ordered_joint_name);
43  _XBotModel.get_enabled_joint_ids_in_chain(chain_name, _ordered_joint_id);
44 
45  // Get urdf model
46  const urdf::ModelInterface &robot_urdf = *XBotModel.get_urdf_model();
47 
48  // resize joint vector
49  _joint_vector.resize(_joint_num);
50  // initialize the joint maps and vector
51  for (int i = 0; i < _joint_num; i++) {
52  std::string actual_joint_name = _ordered_joint_name[i];
53  int actual_joint_id = _ordered_joint_id[i];
54  XBot::Joint::Ptr actual_joint = std::make_shared<Joint>(actual_joint_name,
55  actual_joint_id,
56  robot_urdf.getJoint(actual_joint_name),
57  _chain_name);
58  _joint_name_map[actual_joint_name] = actual_joint;
59  _joint_id_map[actual_joint_id] = actual_joint;
60  _joint_vector[i] = actual_joint;
61  }
62 
63 
64  // Fill _urdf_joints and _urdf_links
65  // NOTE: is it the correct way to handle links? What happens if some joint
66  // along the chain is disabled/fixed/...?
67  for (const std::string & joint_name : _ordered_joint_name) {
68 
69  _urdf_joints.push_back(robot_urdf.getJoint(joint_name));
70 
71  const std::string &parent_link_name = robot_urdf.getJoint(joint_name)->parent_link_name;
72  _urdf_links.push_back(robot_urdf.getLink(parent_link_name));
73 
74  }
75 
76  // Add last link to _urdf_links
77  _urdf_links.push_back(robot_urdf.getLink(getChildLinkName(getJointNum() - 1)));
78 
79  // Add FT
80  for( const auto& ft_name_id : _XBotModel.get_ft_sensors() ){
81 
82  const std::string& ft_joint_name = ft_name_id.first;
83  const int ft_id = ft_name_id.second;
84  std::string ft_link_name = robot_urdf.getJoint(ft_joint_name)->child_link_name;
85  const std::string& ft_parent_link_name = robot_urdf.getJoint(ft_joint_name)->parent_link_name;
86 
87  // check the FT on this chain
88  for( const auto& link_in_chain : _urdf_links) {
89  if(link_in_chain->name == ft_parent_link_name) {
90  ForceTorqueSensor::Ptr ft_ptr = std::make_shared<ForceTorqueSensor>(robot_urdf.getLink(ft_link_name), ft_id);
91  _ft_vector.push_back(ft_ptr);
92  _ft_map[ft_ptr->getSensorName()] = ft_ptr;
93  std::cout << "****************** FOUND FT " << ft_link_name << " ON CHAIN " << getChainName() << std::endl;
94  }
95  }
96 
97 
98  }
99 
100  // Add IMU
101  for( const auto& imu_name_id : _XBotModel.get_imu_sensors() ){
102 
103  const int imu_id = imu_name_id.second;
104  std::string imu_link_name = imu_name_id.first;
105  std::string imu_parent_link_name =
106  robot_urdf.getLink(imu_link_name)->parent_joint->parent_link_name;
107 
108  // check the if IMU is on this chain
109  for( const auto& link_in_chain : _urdf_links) {
110  if(link_in_chain->name == imu_parent_link_name) {
111  ImuSensor::Ptr imu_ptr = std::make_shared<ImuSensor>(robot_urdf.getLink(imu_link_name), imu_id);
112  _imu_vector.push_back(imu_ptr);
113  _imu_map[imu_ptr->getSensorName()] = imu_ptr;
114  std::cout << "****************** FOUND IMU " << imu_link_name << " ON CHAIN " << getChainName() << std::endl;
115  }
