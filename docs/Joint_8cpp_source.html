295  tp.AddColumn("Effort", 8);
296  tp.AddColumn("Tau ref", 7);

301  tp.AddColumn("Vel err", 7);
302  tp.AddColumn("Vel err%", 8);
303 
304  double pos_err = getPositionReference() - getJointPosition();
305  double pos_err_rel = pos_err / getPositionReference();
306  double tau_err = getEffortReference() - getJointEffort();
307  double tau_err_rel = pos_err / getEffortReference();
308  double vel_err = getVelocityReference() - getJointVelocity();
309  double vel_err_rel = pos_err / getVelocityReference();
310 
311  tp.PrintHeader();
312  tp << getJointName() << getJointId() << getJointPosition() << getPositionReference() << pos_err << pos_err_rel*100 << getJointEffort() << getEffortReference() << tau_err << tau_err_rel*100 << getJointVelocity() << getVelocityReference() << vel_err << vel_err_rel*100;
313  tp.PrintFooter();
314 }
