274 template <typename... SyncFlags>
275 bool XBot::RobotInterface::setReferenceFrom ( const XBot::ModelInterface& model, SyncFlags... flags )
276 {
277  bool success = true;
278  for (const auto & c : model._model_chain_map) {
279 
280  const std::string &chain_name = c.first;
281  const ModelChain &chain = *c.second;
282 
283  if (_robot_chain_map.count(chain_name)) {
284  _robot_chain_map.at(chain_name)->setReferenceFrom(chain, flags...);
285  } else {
286  if(!chain.isVirtual()){
287  std::cerr << "ERROR " << __func__ << " : you are trying to synchronize XBotInterfaces with different chains!!" << std::endl;
288  success = false;
289  }
290  }
291  }
292  return success;
293 }
