47  void open(const filename_t& fname, bool truncate = false)
48  {
49 
50  close();
51  auto *mode = truncate ? SPDLOG_FILENAME_T("wb") : SPDLOG_FILENAME_T("ab");
52  _filename = fname;
53  for (int tries = 0; tries < open_tries; ++tries)
