101 // An argument visitor that converts an integer argument to T for printf,
102 // if T is an integral type. If T is void, the argument is converted to

116  : arg_(arg), type_(type)

119  void visit_bool(bool value)
120  {
121  if (type_ != 's')
122  visit_any_int(value);
123  }

125  template <typename U>
126  void visit_any_int(U value)
127  {
128  bool is_signed = type_ == 'd' || type_ == 'i';
129  using internal::Arg;
130  typedef typename internal::Conditional<
131  is_same<T, void>::value, U, T>::type TargetType;
132  if (sizeof(TargetType) <= sizeof(int))
133  {
134  // Extra casts are used to silence warnings.
135  if (is_signed)
136  {
137  arg_.type = Arg::INT;
138  arg_.int_value = static_cast<int>(static_cast<TargetType>(value));
139  }
140  else
141  {
142  arg_.type = Arg::UINT;
143  typedef typename internal::MakeUnsigned<TargetType>::Type Unsigned;
144  arg_.uint_value = static_cast<unsigned>(static_cast<Unsigned>(value));
145  }
146  }
147  else
148  {
149  if (is_signed)
150  {
151  arg_.type = Arg::LONG_LONG;
152  // glibc's printf doesn't sign extend arguments of smaller types:
153  // std::printf("%lld", -42); // prints "4294967254"
154  // but we don't have to do the same because it's a UB.
155  arg_.long_long_value = static_cast<LongLong>(value);
156  }
157  else
158  {
159  arg_.type = Arg::ULONG_LONG;
160  arg_.ulong_long_value =
161  static_cast<typename internal::MakeUnsigned<U>::Type>(value);
162  }
163  }
164  }

422 template <typename Char, typename AF>
423 unsigned PrintfFormatter<Char, AF>::parse_header(
424  const Char *&s, FormatSpec &spec)
425 {
426  unsigned arg_index = std::numeric_limits<unsigned>::max();
427  Char c = *s;
428  if (c >= '0' && c <= '9')
429  {
430  // Parse an argument index (if followed by '$') or a width possibly
431  // preceded with '0' flag(s).
432  unsigned value = internal::parse_nonnegative_int(s);
433  if (*s == '$') // value is an argument index
434  {
435  ++s;
436  arg_index = value;
437  }
438  else
439  {
440  if (c == '0')
441  spec.fill_ = '0';
442  if (value != 0)
443  {
444  // Nonzero value means that we parsed width and don't need to
445  // parse it or flags again, so return now.
446  spec.width_ = value;
447  return arg_index;
448  }
449  }
450  }
451  parse_flags(spec, s);
452  // Parse width.
453  if (*s >= '0' && *s <= '9')
454  {
455  spec.width_ = internal::parse_nonnegative_int(s);
456  }
457  else if (*s == '*')
458  {
459  ++s;
460  spec.width_ = internal::WidthHandler(spec).visit(get_arg(s));
461  }
462  return arg_index;
463 }

465 template <typename Char, typename AF>
466 void PrintfFormatter<Char, AF>::format(BasicCStringRef<Char> format_str)
467 {
468  const Char *start = format_str.c_str();
469  const Char *s = start;
470  while (*s)
471  {
472  Char c = *s++;
473  if (c != '%') continue;
474  if (*s == c)
475  {
476  write(writer_, start, s);
477  start = ++s;
478  continue;
479  }
480  write(writer_, start, s - 1);
481 
482  FormatSpec spec;
483  spec.align_ = ALIGN_RIGHT;
484 
485  // Parse argument index, flags and width.
486  unsigned arg_index = parse_header(s, spec);
487 
488  // Parse precision.
489  if (*s == '.')
490  {
491  ++s;
492  if ('0' <= *s && *s <= '9')
493  {
494  spec.precision_ = static_cast<int>(internal::parse_nonnegative_int(s));
495  }
496  else if (*s == '*')
497  {
498  ++s;
499  spec.precision_ = internal::PrecisionHandler().visit(get_arg(s));
500  }
501  }
502 
503  using internal::Arg;
504  Arg arg = get_arg(s, arg_index);
505  if (spec.flag(HASH_FLAG) && internal::IsZeroInt().visit(arg))
506  spec.flags_ &= ~internal::to_unsigned<int>(HASH_FLAG);
507  if (spec.fill_ == '0')
508  {
509  if (arg.type <= Arg::LAST_NUMERIC_TYPE)
510  spec.align_ = ALIGN_NUMERIC;
511  else
512  spec.fill_ = ' '; // Ignore '0' flag for non-numeric types.
513  }
514 
515  // Parse length and convert the argument to the required type.
516  using internal::ArgConverter;
517  switch (*s++)
518  {
519  case 'h':
520  if (*s == 'h')
521  ArgConverter<signed char>(arg, *++s).visit(arg);
522  else
523  ArgConverter<short>(arg, *s).visit(arg);
524  break;
525  case 'l':
526  if (*s == 'l')
527  ArgConverter<fmt::LongLong>(arg, *++s).visit(arg);
528  else
529  ArgConverter<long>(arg, *s).visit(arg);
530  break;
531  case 'j':
532  ArgConverter<intmax_t>(arg, *s).visit(arg);
533  break;
534  case 'z':
535  ArgConverter<std::size_t>(arg, *s).visit(arg);
536  break;
537  case 't':
538  ArgConverter<std::ptrdiff_t>(arg, *s).visit(arg);
539  break;
540  case 'L':
541  // printf produces garbage when 'L' is omitted for long double, no
542  // need to do the same.
543  break;
544  default:
545  --s;
546  ArgConverter<void>(arg, *s).visit(arg);
547  }
548 
549  // Parse type.
550  if (!*s)
551  FMT_THROW(FormatError("invalid format string"));
552  spec.type_ = static_cast<char>(*s++);
553  if (arg.type <= Arg::LAST_INTEGER_TYPE)
554  {
555  // Normalize type.
556  switch (spec.type_)
557  {
558  case 'i':
559  case 'u':
560  spec.type_ = 'd';
561  break;
562  case 'c':
563  // TODO: handle wchar_t
564  internal::CharConverter(arg).visit(arg);
565  break;
566  }
567  }
568 
569  start = s;
570 
571  // Format argument.
572  AF(writer_, spec).visit(arg);
573  }
574  write(writer_, start, s);
575 }
