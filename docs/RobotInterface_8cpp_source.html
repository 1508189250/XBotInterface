37 bool XBot::RobotInterface::parseYAML(const std::string &path_to_cfg, const std::string& framework)
38 {
39  std::ifstream fin(path_to_cfg);
40  if (fin.fail()) {
41  std::cerr << "ERROR in " << __func__ << "! Can NOT open " << path_to_cfg << "!" << std::endl;
42  return false;
43  }
44 
45  // loading YAML
46  YAML::Node root_cfg = YAML::LoadFile(path_to_cfg);
47  YAML::Node x_bot_interface;
48  // XBotInterface info
49  if(root_cfg["RobotInterface"]) {
50  x_bot_interface = root_cfg["RobotInterface"];
51  }
52  else {
53  std::cerr << "ERROR in " << __func__ << " : YAML file " << path_to_cfg << " does not contain RobotInterface mandatory node!!" << std::endl;
54  return false;
55  }
56  // check framework
57  if(x_bot_interface["framework_name"]) {
58  _framework = x_bot_interface["framework_name"].as<std::string>();
59  }
60  else {
61  std::cerr << "ERROR in " << __func__ << " : RobotInterface node of " << path_to_cfg << " does not contain framework_name mandatory node!!" << std::endl;
62  return false;
63  }
64 
65  if( framework != "" ){
66  _framework = framework;
67  }
68 
69  // subclass forced
70  _subclass_name = std::string("RobotInterface") + _framework;
71  // check the path to shared lib
72  if(root_cfg[_subclass_name]["path_to_shared_lib"]) {
73  computeAbsolutePath(root_cfg[_subclass_name]["path_to_shared_lib"].as<std::string>(),
74  LIB_MIDDLE_PATH,
75  _path_to_shared_lib);
76  }
77  else {
78  std::cerr << "ERROR in " << __func__ << " : YAML file " << path_to_cfg << " does not contain " << _subclass_name << " mandatory node!!" << std::endl;
79  return false;
80  }
81 
82  if(root_cfg[_subclass_name]["subclass_factory_name"]) {
83  _subclass_factory_name = root_cfg[_subclass_name]["subclass_factory_name"].as<std::string>();
84  }
85  else {
86  std::cerr << "ERROR in " << __func__ << " : " << _subclass_name << " node of " << path_to_cfg << " does not contain subclass_factory_name mandatory node!!" << std::endl;
87  return false;
88  }
89  return true;

121  // open and init robot interface

123  _robot_interface_instance->init(path_to_cfg, any_map);
124  // static instance of the robot interface

159 bool XBot::RobotInterface::init_internal(const std::string& path_to_cfg, AnyMapConstPtr any_map)
160 {
161  // Fill _robot_chain_map with shallow copies of chains in _chain_map
162 
163  _robot_chain_map.clear();
164 
165  for( const auto& c : _chain_map){
166 
167  RobotChain::Ptr robot_chain(new RobotChain());
168  robot_chain->shallowCopy(*c.second);
169 
170  _robot_chain_map[c.first] = robot_chain;
171 
172  }
173 
174  // Call virtual init_robot
175  bool success = init_robot(path_to_cfg, any_map);
176 
177  _ordered_chain_names.clear();
178  for( const std::string& s : model().getModelOrderedChainName() ){
179  if( s == "virtual_chain" ){}
180  else{
181  _ordered_chain_names.push_back(s);
182  }
183  }
184 
185  // Since fixed controlled joints do not appear in the model, some chains may
186  // be missing. So we add them!
187  for( const auto& pair : _chain_map ){
188  auto it = std::find(_ordered_chain_names.begin(), _ordered_chain_names.end(), pair.first);
189  if( it == _ordered_chain_names.end() ){
190  _ordered_chain_names.push_back(pair.first);
191  }
192  }
193 
194  return success;
195 }
